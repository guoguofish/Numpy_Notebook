greater\(x, y \[,out\]\)      x &gt; y

greater\_equal\(x,  y  \[,out\]\)    x &gt;= y

less\(x, y \[,out\]\)      x &lt; y

less\_equal\(x, y \[,out\]\)      x &lt;= y

not\_equal\( x, y \[,out\]\)    $$x \neq y$$

equal\(x,  y \[,out\]\)   x==y
